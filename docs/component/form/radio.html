    # 单选框 y-radio

    在一组备选项中进行单选

    # 组件名称

    1. y-radio
    2. y-radio-group

    # 基础用法

    # 单选框组

    # radio属性

    export type IModelType = boolean | string | number

/**
 * 属性
 */
export interface Props {

    /**
     * 选中值
     */
    modelValue: IModelType

    /**
     * 值
     */
    value: IModelType

    /**
     * 是否不可选，默认false
     */
    disabled?: boolean
}
