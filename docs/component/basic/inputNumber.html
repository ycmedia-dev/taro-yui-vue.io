    # 计数器 y-input-number

    计数器，支持编辑/不可编辑

    # 组件名称

    y-input-number

    # 基础用法

    # 不可编辑

    # 属性

    /**
 * 属性
 */
export interface YInputNumberProps extends Record<string, unknown> {
  /**
   * 数目，值
   */
  modelValue: number

  /***
   * 组件参数
   */
  options: InputNumberInfo
}

export interface InputNumberInfo extends Record<string, unknown> {
  /***
   * 是否可编辑,false:可编辑， true: 不可编辑，默认false
   */
  disabled?: boolean

  /**
   * 最大: 默认9999
   */
  maxCount: number

  /**
   * 最小： 默认1
   */
  minCount: number
}
