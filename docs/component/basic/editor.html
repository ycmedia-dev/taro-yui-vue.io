export interface YEditorProps extends Record<string, unknown> {
  /**
   * 值
   */
  modelValue: string

  /**
   * 组件信息
   */
  options: EditorOptions
}

export interface EditorOptions extends Record<string, unknown> {
  /**
   * placeholder
   */
  placeholder: string

  /**
   * 编辑器功能配置
   * */
  toolList: string[]

  /**
   * 是否仅读， 默认false
   */
  readOnly: boolean
}
